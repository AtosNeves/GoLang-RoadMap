Context

Controle de Requisições e Cancelamento

O que é Context?

O package context é usado para controlar cancelamento, timeouts e propagação de valores através de chamadas de API e goroutines.

Casos de Uso Principais:
  • Cancelar operações quando não são mais necessárias
  • Definir timeouts para operações
  • Passar valores específicos da requisição (como IDs de trace)

Context Básico

context_basico.go 
package main

import (
    "context"
    "fmt"
    "time"
)

func operacao(ctx context.Context, nome string) {
    select {
    case <-time.After(2 * time.Second):
        fmt.Printf("%s concluída\n", nome)
    case <-ctx.Done():
        fmt.Printf("%s cancelada: %v\n", nome, ctx.Err())
    }
}

func main() {
    // Context com cancelamento manual
    ctx, cancel := context.WithCancel(context.Background())
    
    go operacao(ctx, "Operação 1")
    
    time.Sleep(1 * time.Second)
    cancel() // Cancela a operação
    
    time.Sleep(2 * time.Second)
}

Tipos de Context

1. WithTimeout

with_timeout.go 
package main

import (
    "context"
    "fmt"
    "time"
)

func buscarDados(ctx context.Context) (string, error) {
    // Simula operação demorada
    select {
    case <-time.After(3 * time.Second):
        return "Dados recuperados", nil
    case <-ctx.Done():
        return "", ctx.Err()
    }
}

func main() {
    // Timeout de 2 segundos
    ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
    defer cancel()
    
    dados, err := buscarDados(ctx)
    if err != nil {
        fmt.Println("Erro:", err) // context deadline exceeded
        return
    }
    
    fmt.Println(dados)
}

2. WithDeadline

with_deadline.go 
package main

import (
    "context"
    "fmt"
    "time"
)

func main() {
    // Deadline específico
    deadline := time.Now().Add(2 * time.Second)
    ctx, cancel := context.WithDeadline(context.Background(), deadline)
    defer cancel()
    
    select {
    case <-time.After(3 * time.Second):
        fmt.Println("Operação concluída")
    case <-ctx.Done():
        fmt.Println("Deadline atingido:", ctx.Err())
    }
}

3. WithValue

with_value.go 
package main

import (
    "context"
    "fmt"
)

type contextKey string

const (
    userIDKey    contextKey = "userID"
    requestIDKey contextKey = "requestID"
)

func processarRequisicao(ctx context.Context) {
    userID := ctx.Value(userIDKey).(string)
    requestID := ctx.Value(requestIDKey).(string)
    
    fmt.Printf("Processando - UserID: %s, RequestID: %s\n", userID, requestID)
}

func main() {
    ctx := context.Background()
    ctx = context.WithValue(ctx, userIDKey, "user-123")
    ctx = context.WithValue(ctx, requestIDKey, "req-456")
    
    processarRequisicao(ctx)
}

Exemplos Práticos

1. Requisição HTTP com Context

http_context.go 
package main

import (
    "context"
    "fmt"
    "io"
    "net/http"
    "time"
)

func fazerRequisicao(ctx context.Context, url string) (string, error) {
    req, err := http.NewRequestWithContext(ctx, "GET", url, nil)
    if err != nil {
        return "", err
    }
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()
    
    body, err := io.ReadAll(resp.Body)
    return string(body), err
}

func main() {
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    
    dados, err := fazerRequisicao(ctx, "https://api.example.com/data")
    if err != nil {
        fmt.Println("Erro:", err)
        return
    }
    
    fmt.Println("Dados:", dados)
}

2. Worker Pool com Context

worker_pool_context.go 
package main

import (
    "context"
    "fmt"
    "sync"
    "time"
)

func worker(ctx context.Context, id int, jobs <-chan int, wg *sync.WaitGroup) {
    defer wg.Done()
    
    for {
        select {
        case job, ok := <-jobs:
            if !ok {
                return
            }
            fmt.Printf("Worker %d processando job %d\n", id, job)
            time.Sleep(500 * time.Millisecond)
        case <-ctx.Done():
            fmt.Printf("Worker %d cancelado\n", id)
            return
        }
    }
}

func main() {
    ctx, cancel := context.WithTimeout(context.Background(), 3*time.Second)
    defer cancel()
    
    jobs := make(chan int, 10)
    var wg sync.WaitGroup
    
    // Inicia workers
    for w := 1; w <= 3; w++ {
        wg.Add(1)
        go worker(ctx, w, jobs, &wg)
    }
    
    // Envia jobs
    for j := 1; j <= 10; j++ {
        jobs <- j
    }
    close(jobs)
    
    wg.Wait()
    fmt.Println("Todos workers finalizados")
}

Boas Práticas

Faça

  • Sempre chame cancel() usando defer após criar um context
  • Passe context como primeiro parâmetro das funções
  • Use context.Background() no ponto de entrada da aplicação
  • Use context.TODO() quando não tiver certeza qual context usar
  • Respeite o cancelamento em operações de longa duração

Não Faça

  • Não armazene context em structs
  • Não use context.Value para passar parâmetros obrigatórios
  • Não ignore context.Done() em loops ou operações demoradas
  • Não crie contexts desnecessários em operações rápidas

Context é essencial para controlar o ciclo de vida de operações em Go
