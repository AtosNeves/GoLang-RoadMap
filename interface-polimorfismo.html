O que são Interfaces?

Uma interface em Go é um conjunto de assinaturas de métodos. Ela define um contrato de comportamento sem especificar como esse comportamento deve ser implementado.

Conceito Chave: Em Go, interfaces são implementadas implicitamente. Se um tipo possui todos os métodos que uma interface define, ele automaticamente implementa essa interface - sem declaração explícita!

Declaração Básica

interface_basica.go 
package main

import "fmt"

// Declarando uma interface
type Forma interface {
    Area() float64
    Perimetro() float64
}

// Retângulo implementa Forma
type Retangulo struct {
    Largura, Altura float64
}

func (r Retangulo) Area() float64 {
    return r.Largura * r.Altura
}

func (r Retangulo) Perimetro() float64 {
    return 2 * (r.Largura + r.Altura)
}

// Círculo implementa Forma
type Circulo struct {
    Raio float64
}

func (c Circulo) Area() float64 {
    return 3.14159 * c.Raio * c.Raio
}

func (c Circulo) Perimetro() float64 {
    return 2 * 3.14159 * c.Raio
}

func main() {
    var f Forma
    
    f = Retangulo{Largura: 10, Altura: 5}
    fmt.Printf("Retângulo - Área: %.2f, Perímetro: %.2f\n", f.Area(), f.Perimetro())
    
    f = Circulo{Raio: 7}
    fmt.Printf("Círculo - Área: %.2f, Perímetro: %.2f\n", f.Area(), f.Perimetro())
}

Implementação Implícita

Go não requer palavra-chave como "implements". Se um tipo possui os métodos, ele implementa a interface automaticamente.

implementacao_implicita.go 
package main

import "fmt"

type Escritor interface {
    Escrever(texto string) error
}

type ArquivoLog struct {
    Nome string
}

// ArquivoLog implementa Escritor automaticamente
func (a ArquivoLog) Escrever(texto string) error {
    fmt.Printf("[%s] %s\n", a.Nome, texto)
    return nil
}

type Console struct{}

// Console também implementa Escritor
func (c Console) Escrever(texto string) error {
    fmt.Println("CONSOLE:", texto)
    return nil
}

func registrar(w Escritor, mensagem string) {
    w.Escrever(mensagem)
}

func main() {
    arquivo := ArquivoLog{Nome: "app.log"}
    console := Console{}
    
    registrar(arquivo, "Iniciando aplicação")
    registrar(console, "Processando dados")
}

Interface Vazia

interface_vazia.go 
package main

import "fmt"

// interface{} aceita qualquer tipo
func imprimir(valor interface{}) {
    fmt.Println(valor)
}

func descrever(i interface{}) {
    fmt.Printf("Tipo: %T, Valor: %v\n", i, i)
}

func main() {
    imprimir(42)
    imprimir("Hello")
    imprimir(true)
    imprimir([]int{1, 2, 3})
    
    descrever(3.14)
    descrever("Go")
}
Nota: Go 1.18+ introduziu any como alias para interface{}, tornando o código mais legível.

Polimorfismo

Polimorfismo permite que diferentes tipos sejam tratados através de uma interface comum.

polimorfismo.go 
package main

import "fmt"

type Animal interface {
    EmitirSom() string
    Mover() string
}

type Cachorro struct {
    Nome string
}

func (c Cachorro) EmitirSom() string {
    return "Au au!"
}

func (c Cachorro) Mover() string {
    return "Correndo"
}

type Passaro struct {
    Nome string
}

func (p Passaro) EmitirSom() string {
    return "Piu piu!"
}

func (p Passaro) Mover() string {
    return "Voando"
}

func fazerBarulho(a Animal) {
    fmt.Printf("%s está %s\n", a.EmitirSom(), a.Mover())
}

func main() {
    animais := []Animal{
        Cachorro{Nome: "Rex"},
        Passaro{Nome: "Piu"},
    }
    
    for _, animal := range animais {
        fazerBarulho(animal)
    }
}

Type Assertion e Type Switch

type_assertion.go 
package main

import "fmt"

func processar(i interface{}) {
    // Type assertion com verificação
    if valor, ok := i.(string); ok {
        fmt.Println("String:", valor)
        return
    }
    
    // Type switch
    switch v := i.(type) {
    case int:
        fmt.Printf("Inteiro: %d\n", v)
    case float64:
        fmt.Printf("Float: %.2f\n", v)
    case bool:
        fmt.Printf("Booleano: %t\n", v)
    case []int:
        fmt.Println("Slice de ints:", v)
    default:
        fmt.Printf("Tipo desconhecido: %T\n", v)
    }
}

func main() {
    processar(42)
    processar(3.14)
    processar("Hello")
    processar(true)
    processar([]int{1, 2, 3})
}
