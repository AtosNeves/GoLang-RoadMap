type ContaBancaria struct {

    return &ContaBancaria{
        Numero:      numero,
        Titular:     titular,
        Saldo:       saldoInicial,
        DataCriacao: time.Now(),
    }
}

func (c *ContaBancaria) Depositar(valor float64) error {

        return fmt.Errorf("valor deve ser positivo")

    c.Saldo += valor

    ID    int
    Nome  string

    Produto
    Quantidade int
}

func (i ItemCarrinho) Subtotal() float64 {
    return i.Preco * float64(i.Quantidade)
}

type Carrinho struct {
    Items []ItemCarrinho
}

func (c *Carrinho) Adicionar(produto Produto, quantidade int) {
    // Verifica se produto j√° existe
    for i, item := range c.Items {
        if item.ID == produto.ID {

            fmt.Printf("%s: quantidade atualizada para %d\n", 
                produto.Nome, c.Items[i].Quantidade)
            return
        }
    }
    
    // Adiciona novo item
    c.Items = append(c.Items, ItemCarrinho{
        Produto:    produto,
        Quantidade: quantidade,
    })
    fmt.Printf("%s adicionado ao carrinho\n", produto.Nome)

    for i, item := range c.Items {
        if item.ID == produtoID {
            c.Items = append(c.Items[:i], c.Items[i+1:]...)
            fmt.Println("Produto removido")
            return
        }
    }
}

    total := 0.0
    for _, item := range c.Items {
        total += item.Subtotal()
    }
    return total
}

func (c Carrinho) Exibir() {
    fmt.Println("\n=== CARRINHO DE COMPRAS ===")
    
    if len(c.Items) == 0 {
        fmt.Println("Carrinho vazio")
        return
    }
    
    for _, item := range c.Items {
        fmt.Printf("%s - R$ %.2f x %d = R$ %.2f\n",
            item.Nome, item.Preco, item.Quantidade, item.Subtotal())
    }
    
    fmt.Printf("\nTotal: R$ %.2f\n", c.Total())
}

    p1 := Produto{1, "Mouse", 29.90}
    p2 := Produto{2, "Teclado", 89.90}
    p3 := Produto{3, "Monitor", 599.00}
    
    carrinho.Adicionar(p1, 2)
    carrinho.Adicionar(p2, 1)
    carrinho.Adicionar(p3, 1)
    carrinho.Adicionar(p1, 1)  // Adiciona mais um mouse
    
    carrinho.Exibir()
    
    carrinho.Remover(2)
    carrinho.Exibir()
}
