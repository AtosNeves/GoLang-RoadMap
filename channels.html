O que são Tipos Operações Exemplos

O que são Channels?

Channels são tubos tipados através dos quais você pode enviar e receber valores. Eles permitem comunicação segura entre goroutines.

Filosofia Go: "Não comunique compartilhando memória; compartilhe memória comunicando." Channels implementam essa filosofia.

Criando Channels

channel_basico.go 
package main

import "fmt"

func main() {
    // Criando um channel de inteiros
    ch := make(chan int)
    
    // Enviando valor em goroutine
    go func() {
        ch <- 42  // Envia valor para o channel
    }()
    
    // Recebendo valor
    valor := <-ch  // Recebe valor do channel
    fmt.Println(valor)
    
    // Channels de diferentes tipos
    chString := make(chan string)
    chBool := make(chan bool)
    chStruct := make(chan struct{})
}

Tipos de Channels

1. Unbuffered Channels

unbuffered.go 
package main

import "fmt"

func main() {
    // Channel sem buffer - bloqueante
    ch := make(chan string)
    
    go func() {
        // Envia e bloqueia até que alguém receba
        ch <- "Mensagem"
        fmt.Println("Enviado!")
    }()
    
    // Recebe e bloqueia até que alguém envie
    msg := <-ch
    fmt.Println("Recebido:", msg)
}

2. Buffered Channels

buffered.go 
package main

import "fmt"

func main() {
    // Channel com buffer de 2
    ch := make(chan int, 2)
    
    // Pode enviar até 2 valores sem bloquear
    ch <- 1
    ch <- 2
    
    fmt.Println(<-ch)  // 1
    fmt.Println(<-ch)  // 2
    
    // Informações do buffer
    ch <- 10
    fmt.Printf("Capacidade: %d, Tamanho: %d\n", cap(ch), len(ch))
}

3. Directional Channels

directional.go 
package main

import "fmt"

// Apenas envia
func enviar(ch chan<- int) {
    ch <- 100
}

// Apenas recebe
func receber(ch <-chan int) {
    valor := <-ch
    fmt.Println(valor)
}

func main() {
    ch := make(chan int)
    
    go enviar(ch)
    receber(ch)
}

Operações com Channels

Fechando Channels

close_channel.go 
package main

import "fmt"

func main() {
    ch := make(chan int, 3)
    
    // Envia valores
    ch <- 1
    ch <- 2
    ch <- 3
    close(ch)  // Fecha o channel
    
    // Range sobre channel fechado
    for valor := range ch {
        fmt.Println(valor)
    }
    
    // Verificar se channel está fechado
    valor, ok := <-ch
    if !ok {
        fmt.Println("Channel fechado")
    }
}
Importante: Apenas o remetente deve fechar um channel. Enviar para um channel fechado causa panic!

Range sobre Channels

range_channel.go 
package main

import "fmt"

func produtor(ch chan<- int) {
    for i := 0; i < 5; i++ {
        ch <- i
    }
    close(ch)  // Importante!
}

func main() {
    ch := make(chan int)
    
    go produtor(ch)
    
    // Range continua até o channel ser fechado
    for valor := range ch {
        fmt.Println(valor)
    }
}

Exemplos Práticos

1. Pipeline de Dados

pipeline.go 
package main

import "fmt"

func gerador(nums ...int) <-chan int {
    out := make(chan int)
    go func() {
        for _, n := range nums {
            out <- n
        }
        close(out)
    }()
    return out
}

func quadrado(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        for n := range in {
            out <- n * n
        }
        close(out)
    }()
    return out
}

func main() {
    // Pipeline: gerador -> quadrado
    ch := gerador(1, 2, 3, 4, 5)
    resultado := quadrado(ch)
    
    for n := range resultado {
        fmt.Println(n)
    }
}

2. Fan-Out / Fan-In

fan_out_in.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

func worker(id int, jobs <-chan int, results chan<- int) {
    for j := range jobs {
        fmt.Printf("Worker %d processando %d\n", id, j)
        time.Sleep(100 * time.Millisecond)
        results <- j * 2
    }
}

func main() {
    jobs := make(chan int, 10)
    results := make(chan int, 10)
    
    // Fan-out: múltiplos workers
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }
    
    // Envia jobs
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)
    
    // Fan-in: coleta resultados
    for a := 1; a <= 5; a++ {
        fmt.Println(<-results)
    }
}

3. Timeout Pattern

timeout.go 
package main

import (
    "fmt"
    "time"
)

func operacaoDemorada(ch chan<- string) {
    time.Sleep(2 * time.Second)
    ch <- "Resultado"
}

func main() {
    ch := make(chan string)
    
    go operacaoDemorada(ch)
    
    select {
    case resultado := <-ch:
        fmt.Println(resultado)
    case <-time.After(1 * time.Second):
        fmt.Println("Timeout!")
    }
}

4. Quit Channel

quit_channel.go 
package main

import (
    "fmt"
    "time"
)

func worker(id int, quit <-chan bool) {
    for {
        select {
        case <-quit:
            fmt.Printf("Worker %d parando\n", id)
            return
        default:
            fmt.Printf("Worker %d trabalhando\n", id)
            time.Sleep(500 * time.Millisecond)
        }
    }
}

func main() {
    quit := make(chan bool)
    
    go worker(1, quit)
    
    time.Sleep(2 * time.Second)
    quit <- true
    
    time.Sleep(1 * time.Second)
}

Boas Práticas

Faça

  • Sempre feche channels quando não forem mais necessários
  • Feche channels apenas no lado do remetente
  • Use channels direcionais para clareza
  • Use buffered channels para desacoplar produtor/consumidor
  • Use range para iterar sobre channels
  • Considere timeouts para operações que podem travar

Evite

  • Enviar para channel fechado (causa panic)
  • Fechar channel múltiplas vezes (causa panic)
  • Fechar channel no lado receptor
  • Usar channels quando mutex seria mais simples
  • Criar goroutine leaks (goroutines bloqueadas eternamente)

Channels tornam a comunicação entre goroutines segura e elegante
