WaitGroups e Mutex

Sincronização Avançada de Goroutines

WaitGroup

WaitGroup é usado para aguardar que um conjunto de goroutines termine sua execução.

Uso Básico

waitgroup_basico.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

func worker(id int, wg *sync.WaitGroup) {
    defer wg.Done() // Decrementa contador ao finalizar
    
    fmt.Printf("Worker %d iniciando\n", id)
    time.Sleep(time.Second)
    fmt.Printf("Worker %d finalizado\n", id)
}

func main() {
    var wg sync.WaitGroup
    
    for i := 1; i <= 5; i++ {
        wg.Add(1) // Incrementa contador
        go worker(i, &wg)
    }
    
    wg.Wait() // Aguarda contador chegar a zero
    fmt.Println("Todos workers finalizados")
}
Importante: Sempre passe WaitGroup por ponteiro (*sync.WaitGroup). Se passar por valor, cada goroutine terá sua própria cópia e Wait() não funcionará corretamente.

Padrão com Closure

waitgroup_closure.go 
package main

import (
    "fmt"
    "sync"
)

func main() {
    var wg sync.WaitGroup
    urls := []string{
        "https://golang.org",
        "https://google.com",
        "https://github.com",
    }
    
    for _, url := range urls {
        wg.Add(1)
        go func(u string) { // Captura url como parâmetro
            defer wg.Done()
            fmt.Printf("Fetching %s\n", u)
        }(url)
    }
    
    wg.Wait()
}

Mutex

Mutex (Mutual Exclusion) garante que apenas uma goroutine acesse um recurso por vez.

Problema sem Mutex (Race Condition)

race_condition.go 
package main

import (
    "fmt"
    "sync"
)

var contador = 0

func incrementar(wg *sync.WaitGroup) {
    defer wg.Done()
    contador++ // Race condition!
}

func main() {
    var wg sync.WaitGroup
    
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go incrementar(&wg)
    }
    
    wg.Wait()
    fmt.Println("Contador:", contador) // Valor imprevisível!
}

Solução com Mutex

mutex_solucao.go 
package main

import (
    "fmt"
    "sync"
)

type Contador struct {
    mu    sync.Mutex
    valor int
}

func (c *Contador) Incrementar() {
    c.mu.Lock()   // Bloqueia acesso
    c.valor++
    c.mu.Unlock() // Libera acesso
}

func (c *Contador) Valor() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.valor
}

func main() {
    contador := &Contador{}
    var wg sync.WaitGroup
    
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            contador.Incrementar()
        }()
    }
    
    wg.Wait()
    fmt.Println("Contador:", contador.Valor()) // 1000 garantido!
}

RWMutex

RWMutex permite múltiplas leituras simultâneas, mas apenas uma escrita por vez.

rwmutex.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}

func NewCache() *Cache {
    return &Cache{
        data: make(map[string]string),
    }
}

func (c *Cache) Get(key string) (string, bool) {
    c.mu.RLock()         // Read lock (múltiplos leitores)
    defer c.mu.RUnlock()
    val, ok := c.data[key]
    return val, ok
}

func (c *Cache) Set(key, value string) {
    c.mu.Lock()           // Write lock (exclusivo)
    defer c.mu.Unlock()
    c.data[key] = value
}

func main() {
    cache := NewCache()
    var wg sync.WaitGroup
    
    // 1 escritor
    wg.Add(1)
    go func() {
        defer wg.Done()
        for i := 0; i < 10; i++ {
            cache.Set(fmt.Sprintf("key%d", i), "valor")
            time.Sleep(100 * time.Millisecond)
        }
    }()
    
    // 5 leitores simultâneos
    for i := 0; i < 5; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for j := 0; j < 10; j++ {
                cache.Get("key0")
                time.Sleep(50 * time.Millisecond)
            }
        }(i)
    }
    
    wg.Wait()
}

Exemplos Práticos

1. Pool de Conexões Thread-Safe

connection_pool.go 
package main

import (
    "errors"
    "fmt"
    "sync"
)

type ConnectionPool struct {
    mu          sync.Mutex
    connections []int
    maxSize     int
}

func NewConnectionPool(maxSize int) *ConnectionPool {
    return &ConnectionPool{
        connections: make([]int, 0, maxSize),
        maxSize:     maxSize,
    }
}

func (p *ConnectionPool) Get() (int, error) {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    if len(p.connections) == 0 {
        return 0, errors.New("no connections available")
    }
    
    conn := p.connections[len(p.connections)-1]
    p.connections = p.connections[:len(p.connections)-1]
    return conn, nil
}

func (p *ConnectionPool) Put(conn int) {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    if len(p.connections) < p.maxSize {
        p.connections = append(p.connections, conn)
    }
}

func main() {
    pool := NewConnectionPool(5)
    
    // Adiciona conexões
    for i := 1; i <= 5; i++ {
        pool.Put(i)
    }
    
    var wg sync.WaitGroup
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            conn, err := pool.Get()
            if err != nil {
                fmt.Printf("Worker %d: %v\n", id, err)
                return
            }
            fmt.Printf("Worker %d usando conexão %d\n", id, conn)
            pool.Put(conn)
        }(i)
    }
    
    wg.Wait()
}

2. Atomic Operations

atomic.go 
package main

import (
    "fmt"
    "sync"
    "sync/atomic"
)

func main() {
    var contador int64
    var wg sync.WaitGroup
    
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            atomic.AddInt64(&contador, 1) // Thread-safe sem mutex
        }()
    }
    
    wg.Wait()
    fmt.Println("Contador:", atomic.LoadInt64(&contador))
}

Boas Práticas

WaitGroup

  • Sempre passe WaitGroup por ponteiro
  • Chame Add() antes de iniciar a goroutine
  • Use defer para garantir que Done() seja chamado
  • Nunca copie um WaitGroup após o primeiro uso

Mutex

  • Use defer para garantir Unlock()
  • Mantenha seções críticas pequenas
  • Nunca bloqueie por muito tempo
  • Prefira RWMutex quando leituras são frequentes
  • Use atomic para operações simples em números

Evite Deadlocks

  • Nunca chame Lock() duas vezes sem Unlock()
  • Sempre use defer para Unlock()
  • Evite locks aninhados quando possível
  • Documente ordem de locks se necessário

Sincronização adequada é essencial para programação concorrente segura
